Function

Core combinators, composition, Reader monad, and error handling.

id :: a -> a

Identity – returns its argument unchanged.

id (42) // => 42

always :: a -> b -> a

Returns a function that always returns the same value, ignoring its argument.

always (42) ('ignored') // => 42

thrush :: a -> (a -> b) -> b

Thrush / reverse-application — applies an argument to a function.
Useful for point-free pipelines where the value leads.

thrush (42) (x => x + 1) // => 43

tap :: (a -> *) -> a -> a

Runs a side-effect with the value and returns the value unchanged.

tap (console.log) (42) // logs 42, returns 42

flip :: (a -> b -> c) -> b -> a -> c

Flips the order of the first two arguments of a curried binary function.

flip (a => b => a - b) (1) (3) // => 2

on :: (b -> b -> c) -> (a -> b) -> a -> a -> c

P combinator — applies a curried binary function after mapping both
arguments through the same unary function.

on (a => b => a + b) (x => x * 2) (3) (4) // => 14

once :: (a -> b) -> a -> b

Calls a function at most once; all subsequent calls return the first result.

const f = once (x => x + 1)
f (1) // => 2
f (99) // => 2  (cached)

memoize :: (a -> b) -> a -> b

Memoises a unary function using a Map keyed by the argument.

const fib = memoize (n => n <= 1 ? n : fib (n - 1) + fib (n - 2))
fib (10) // => 55

allPass :: (a -> Boolean) -> (a -> Boolean) -> a -> Boolean

Returns a predicate that is true only when both predicates hold (&&).
Equivalent to Haskell's (&&) lifted over predicates.

allPass (x => x > 0) (x => x < 10) (5) // => true
allPass (x => x > 0) (x => x < 10) (0) // => false

anyPass :: (a -> Boolean) -> (a -> Boolean) -> a -> Boolean

Returns a predicate that is true when at least one predicate holds (||).
Equivalent to Haskell's (||) lifted over predicates.

anyPass (x => x < 0) (x => x > 10) (15) // => true
anyPass (x => x < 0) (x => x > 10) (5)  // => false

pipe :: Array (Any -> Any) -> a -> b

Threads a value left-to-right through an array of functions.

pipe ([x => x + 1, Math.sqrt]) (99) // => 10

pipeWith :: (f (a->b) -> f a -> f b) -> f a -> Array (a -> f b) -> f b

Threads a value through a sequence of steps where each step is applied via
a custom "bind" combinator. The canonical use-case is ADT pipelines:
pass the ADT's `chain` (or `map`) as the binder so each step can
short-circuit or transform the wrapped value without manual unwrapping.

pipeWith (chain) (just (1)) ([x => just (x + 1), x => just (x * 2)])
is equivalent to:
chain (x => just (x * 2)) (chain (x => just (x + 1)) (just (1)))


// Maybe pipeline — short-circuits on Nothing
import * as M from './maybe.js'
pipeWith (M.chain) (M.just (4)) ([
  x => x > 0  ? M.just (x)      : M.nothing (),
  x => x < 10 ? M.just (x * 2)  : M.nothing (),
])
// => just(8)

// map-only pipeline (no short-circuiting)
pipeWith (M.map) (M.just (3)) ([x => x + 1, x => x * 2])
// => just(8)

compose :: (b -> c) -> (a -> b) -> a -> c

Right-to-left composition of two functions.
For longer chains, prefer `pipe`.

compose (x => x * 2) (x => x + 1) (3) // => 8

juxt :: Array (a -> b) -> a -> Array b

Applies a single value to each function in the array and collects results.

juxt ([x => x + 1, x => x * 2]) (3) // => [4, 6]

converge :: (b -> c -> d) -> (a -> b) -> (a -> c) -> a -> d

Converge / fork — maps a value through two functions and combines the
results with a binary function. Useful for point-free "split-merge" style.

converge (x => y => x + y) (x => x * 2) (x => x + 1) (3) // => 10

uncurry :: (a -> b -> c) -> [a, b] -> c

Converts a curried binary function to one that accepts a 2-element array.

uncurry (a => b => a + b) ([1, 2]) // => 3

curry :: ([a, b] -> c) -> a -> b -> c

Converts a function that accepts a 2-element array to a curried binary function.

curry (([a, b]) => a + b) (1) (2) // => 3

map :: (a -> b) -> (e -> a) -> e -> b

Reader Functor — post-composes f over a reader.
`map (f) (g)` ≡ `compose (f) (g)`

map (x => x + 1) (x => x * 2) (3) // => 7

ap :: (e -> a -> b) -> (e -> a) -> e -> b

Reader Applicative — S combinator.
`ap (ff) (fa) (x)` = `ff (x) (fa (x))`

ap (e => x => e + x) (e => e * 2) (3) // => 9

of :: a -> e -> a

Reader Applicative — lifts a constant value into the Reader context.
Identical to `always`.

of (42) ('ignored') // => 42

chain :: (a -> e -> b) -> (e -> a) -> e -> b

Reader Monad bind — threads the environment through both readers.

chain (a => e => a + e) (e => e * 2) (3) // => 9

contramap :: (b -> a) -> (a -> c) -> b -> c

Contravariant map — pre-composes a function over a reader's input.

contramap (x => x + 1) (x => x * 2) (3) // => 8

promap :: (a -> b) -> (c -> d) -> (b -> c) -> a -> d

Profunctor map — contramaps the input and maps the output.

promap (x => x + 1) (x => x * 2) (x => x) (3) // => 8

extend :: (e -> e -> e) -> ((e -> a) -> b) -> (e -> a) -> e -> b

Reader Comonad extend — widens the environment by accumulating it with
a semigroup-like concat function.

extend (x => y => x + y) (f => f (0)) (e => e * 2) (3) // => 6

pipeK :: Array (a -> e -> b) -> (e -> a) -> e -> b

Left-to-right Kleisli composition over the Reader monad.
Each function in `fns` must have type `a -> e -> b`.

pipeK ([f, g]) (mx) // chain (g) (chain (f) (mx))

chainRec :: ((a -> Step, b -> Step, a) -> e -> Step) -> a -> e -> b

Stack-safe tail-recursive Reader bind (Fantasy Land ChainRec).

chainRec ((next, done, n) => _ => n <= 0 ? done (n) : next (n - 1)) (1000) (null)

handleThrow :: ((...a) -> b) -> (b -> a -> r) -> (Error -> a -> r) -> ...a -> r

Wraps a function so that thrown errors are caught and routed to `onThrow`.
Both callbacks receive the same argument list `d` as the original function.

handleThrow (JSON.parse) (r => r) (_ => null) ('{}')  // => {}
handleThrow (JSON.parse) (r => r) (_ => null) ('bad') // => null

Logic

Predicate combinators and control-flow utilities.

complement :: (a -> Boolean) -> a -> Boolean

Returns a predicate that negates the original.

complement (x => x > 0) (-1) // => true

bool :: a -> a -> Boolean -> a

Case analysis on a boolean — returns the false branch or the true branch.

bool ('no') ('yes') (true)  // => 'yes'
bool ('no') ('yes') (false) // => 'no'

ifElse :: (a -> Boolean) -> (a -> b) -> (a -> b) -> a -> b

Applies f if predicate holds, g otherwise.

ifElse (x => x > 0) (x => x) (x => -x) (-3) // => 3

when :: (a -> Boolean) -> (a -> a) -> a -> a

Applies f only when predicate holds, otherwise returns x unchanged.

when (x => x < 0) (() => 0) (-1) // => 0

unless :: (a -> Boolean) -> (a -> a) -> a -> a

Applies f only when predicate does NOT hold, otherwise returns x unchanged.

unless (x => x > 0) (x => -x) (-3) // => 3

cond :: Array [(a -> Boolean), (a -> b)] -> a -> Maybe b

Tries each [pred, fn] pair in order; returns Just(fn(a)) for the first
matching predicate, or Nothing if none match.

cond ([[x => x < 0, x => -x], [x => x > 0, x => x]]) (3) // => just(3)
cond ([[x => x < 0, x => -x]]) (3)                        // => nothing()

condE :: Array [(a -> Boolean), (a -> b)] -> a -> Either a b

Like `cond`, but returns Right(fn(a)) on match and Left(a) when no branch
matches — preserving the original value for further error handling rather
than discarding it.

condE ([[x => x < 0, x => -x], [x => x > 0, x => x]]) (3)  // => right(3)
condE ([[x => x < 0, x => -x]])                         (3)  // => left(3)

condDefault :: (a -> b) -> Array [(a -> Boolean), (a -> b)] -> a -> b

Like `cond`, but takes an explicit default function instead of returning
Maybe — eliminates the need to unwrap Just/Nothing at every call site.

condDefault (x => `unhandled: ${x}`) ([
  [x => x < 0, () => 'negative'],
  [x => x > 0, () => 'positive'],
]) (0)  // => 'unhandled: 0'
condDefault (x => `unhandled: ${x}`) ([
  [x => x < 0, () => 'negative'],
  [x => x > 0, () => 'positive'],
]) (5)  // => 'positive'

match :: Array [(a -> Maybe b), (b -> c)] -> a -> Maybe c

Pattern-matching variant of `cond` — each "predicate" is a function
`a -> Maybe b` that simultaneously tests and extracts data. The first
branch returning Just wins; its inner value is passed to the handler.
This is strictly more powerful than `cond`: use it when the predicate
needs to capture a sub-value (like a regex match or a type-narrowing
accessor) that the handler then consumes.

match ([
  [x => x > 0  ? M.just (x)  : M.nothing (), x => `pos: ${x}`],
  [x => x < 0  ? M.just (-x) : M.nothing (), x => `neg abs: ${x}`],
]) (-5)  // => just('neg abs: 5')
match ([
  [x => x > 0  ? M.just (x)  : M.nothing (), x => `pos: ${x}`],
]) (0)   // => nothing()

condAll :: Array [(a -> Boolean), (a -> b)] -> a -> Array b

Runs every branch whose predicate matches and collects all results —
unlike `cond`, which stops at the first match. Returns an empty array
when no predicate holds.

condAll ([
  [x => x > 0,      () => 'positive'],
  [x => x % 2 === 0, () => 'even'],
  [x => x > 100,    () => 'large'],
]) (4)   // => ['positive', 'even']
condAll ([
  [x => x > 0,      () => 'positive'],
]) (-1)  // => []

guardWith :: Array (a -> Boolean | a -> Maybe b) -> a -> Maybe a

Threads a value through a sequence of guard steps, short-circuiting on the
first Nothing. Each step is either:
- a predicate `a -> Boolean` — Nothing when it returns false, Just(a) otherwise
- a transformer `a -> Maybe b` — its result is passed straight through

The two forms are distinguished automatically: if the step returns a boolean
it is treated as a predicate; if it returns a Maybe (has a `.tag`) it is
treated as a transformer.

guardWith ([
  x => x > 0,              // predicate: filter non-positives
  x => x < 100,            // predicate: filter >= 100
]) (42)   // => just(42)

guardWith ([
  x => x > 0,
  x => x < 100,
]) (-1)   // => nothing()

guardWith ([
  x => x > 0,
  x => M.just (x * 2),     // transformer: doubles the value
  x => x < 100,
]) (42)   // => just(84)

guardWith ([
  x => x > 0,
  x => M.just (x * 2),
  x => x < 100,
]) (60)   // => nothing()  (60 * 2 = 120, fails x < 100)

Maybe

The Maybe monad — Just | Nothing. Represents optional values.

just :: a -> Maybe a

Wraps a value in Just.

just (1) // => { tag: 'just', value: 1 }

nothing :: () -> Maybe a

The empty Maybe — represents absence of a value.

nothing () // => { tag: 'nothing' }

fromNullable :: a -> Maybe a

Returns Just(a) for any non-null/undefined value, Nothing otherwise.

fromNullable (1)         // => just(1)
fromNullable (null)      // => nothing()
fromNullable (undefined) // => nothing()

fromPredicate :: (a -> Boolean) -> a -> Maybe a

Returns Just(a) when the predicate holds, Nothing otherwise.

fromPredicate (x => x > 0) (1)  // => just(1)
fromPredicate (x => x > 0) (-1) // => nothing()

tryCatch :: (() -> a) -> Maybe a

Runs a thunk — Just on success, Nothing if it throws.

tryCatch (() => JSON.parse ('1'))     // => just(1)
tryCatch (() => JSON.parse ('bad'))   // => nothing()

isJust :: a -> Boolean

True when the value is a Just.

isJust (just (1))   // => true
isJust (nothing ()) // => false

isNothing :: a -> Boolean

True when the value is Nothing.

isNothing (nothing ()) // => true
isNothing (just (1))   // => false

maybe :: (() -> b) -> (a -> b) -> Maybe a -> b

Case analysis — the lazy default thunk is called only for Nothing.

maybe (() => 0)  (x => x + 1) (just (1))   // => 2
maybe (() => 0)  (x => x + 1) (nothing ())  // => 0

fromMaybe :: a -> Maybe a -> a

Extracts the value or returns the default.

fromMaybe (0) (just (5))   // => 5
fromMaybe (0) (nothing ()) // => 0

fromMaybe_ :: (() -> a) -> Maybe a -> a

Extracts the value or calls the lazy thunk.

fromMaybe_ (() => 0) (nothing ()) // => 0
fromMaybe_ (() => 0) (just (5))   // => 5

toNull :: Maybe a -> a | null

Extracts the value or returns null.

toNull (just (1))   // => 1
toNull (nothing ()) // => null

toUndefined :: Maybe a -> a | undefined

Extracts the value or returns undefined.

toUndefined (just (1))   // => 1
toUndefined (nothing ()) // => undefined

toEither :: a -> Maybe b -> Either a b

Converts Nothing to Left(def) and Just(v) to Right(v).

toEither ('err') (just (1))   // => right(1)
toEither ('err') (nothing ()) // => left('err')

map :: (a -> b) -> Maybe a -> Maybe b

Applies f to the value inside Just; passes Nothing through.

map (x => x + 1) (just (1))   // => just(2)
map (x => x + 1) (nothing ()) // => nothing()

filter :: (a -> Boolean) -> Maybe a -> Maybe a

Returns Nothing when the predicate fails or the Maybe is already Nothing.

filter (x => x > 0) (just (1))   // => just(1)
filter (x => x > 0) (just (-1))  // => nothing()
filter (x => x > 0) (nothing ()) // => nothing()

ap :: Maybe (a -> b) -> Maybe a -> Maybe b

Applies a Just-wrapped function to a Just-wrapped value.

ap (just (x => x + 1)) (just (1))   // => just(2)
ap (nothing ())         (just (1))   // => nothing()

chain :: (a -> Maybe b) -> Maybe a -> Maybe b

Monadic bind — applies f to Just's value, passes Nothing through.

chain (x => just (x + 1)) (just (1))   // => just(2)
chain (x => just (x + 1)) (nothing ()) // => nothing()

chainNullable :: (a -> b | null | undefined) -> Maybe a -> Maybe b

Maps with a nullable-returning function, lifting the result into Maybe.
Equivalent to chain (compose (fromNullable) (f)).

chainNullable (x => x.name) (just ({ name: 'Alice' })) // => just('Alice')
chainNullable (x => x.name) (just ({}))                // => nothing()

alt :: Maybe a -> Maybe a -> Maybe a

Returns the first Just, falling back to the second.

alt (just (2))  (nothing ()) // => just(2)
alt (nothing ()) (just (1))  // => just(1)
alt (nothing ()) (nothing ()) // => nothing()

fold :: (b -> a -> b) -> b -> Maybe a -> b

Folds a Maybe — returns `init` for Nothing, applies f(init)(value) for Just.

fold (acc => x => acc + x) (0) (just (5))   // => 5
fold (acc => x => acc + x) (0) (nothing ()) // => 0

traverse :: (b -> f b) -> ((a -> b) -> f a -> f b) -> (a -> f b) -> Maybe a -> f (Maybe b)

Sequences a Maybe through an applicative functor.
Accepts explicit apOf / apMap to remain functor-agnostic.

traverse (Array.of) (f => xs => xs.map (f)) (x => [x, -x]) (just (1))   // => [just(1), just(-1)]
traverse (Array.of) (f => xs => xs.map (f)) (x => [x, -x]) (nothing ()) // => [nothing()]

zip :: Maybe a -> Maybe b -> Maybe [a, b]

Combines two Maybes into a Just of a pair, or Nothing if either is Nothing.

zip (just (1)) (just (2))   // => just([1, 2])
zip (nothing ()) (just (2)) // => nothing()

justs :: Array (Maybe a) -> Array a

Collects Just values from an array, discarding Nothings.

justs ([just (1), nothing (), just (2)]) // => [1, 2]

Maps f over an array and collects only the Just results.

mapMaybe (x => x > 0 ? just (x) : nothing ()) ([1, -2, 3]) // => [1, 3]

Either

The Either monad — Left | Right. Represents values with two possibilities.

left :: a -> Either a b

Wraps a value in the Left constructor.

left ('err') // => { tag: 'left', left: 'err' }

right :: b -> Either a b

Wraps a value in the Right constructor.

right (42) // => { tag: 'right', right: 42 }

fromNullable :: (() -> a) -> b -> Either a b

Right when value is non-null/undefined; Left via thunk otherwise.

fromNullable (() => 'missing') (null) // => left('missing')
fromNullable (() => 'missing') (42)   // => right(42)

tryCatch :: ((...a) -> b) -> (Error -> Array a -> c) -> ...a -> Either c b

Runs fn; Right on success, Left via curried onError on throw.

tryCatch (JSON.parse) (e => _ => e.message) ('{"a":1}') // => right({ a: 1 })
tryCatch (JSON.parse) (e => _ => e.message) ('bad')     // => left('...')

fromPredicate :: (a -> Boolean) -> (a -> b) -> a -> Either b a

Right when predicate holds; Left via onFalse otherwise. Fully curried.

fromPredicate (x => x > 0) (x => `${x} is not positive`) (3)  // => right(3)
fromPredicate (x => x > 0) (x => `${x} is not positive`) (-1) // => left('-1 is not positive')

isLeft :: a -> Boolean

True when the value is a Left.

isLeft (left (1))  // => true
isLeft (right (1)) // => false

isRight :: a -> Boolean

True when the value is a Right.

isRight (right (1)) // => true
isRight (left (1))  // => false

isEither :: a -> Boolean

True when the value is either a Left or a Right.

isEither (right (1)) // => true
isEither (left ('e')) // => true
isEither (42)         // => false

either :: (a -> c) -> (b -> c) -> Either a b -> c

Curried case-fold on Either.

either (l => `err: ${l}`) (r => r * 2) (right (21)) // => 42
either (l => `err: ${l}`) (r => r * 2) (left ('x')) // => 'err: x'

fromLeft :: a -> Either a b -> a

Extracts the Left value, or returns the default for Right.

fromLeft ('def') (left ('x'))  // => 'x'
fromLeft ('def') (right (1))   // => 'def'

fromRight :: b -> Either a b -> b

Extracts the Right value, or returns the default for Left.

fromRight (0) (right (42)) // => 42
fromRight (0) (left ('x')) // => 0

fromEither :: Either a a -> a

Extracts the value regardless of constructor when both sides share a type.

fromEither (left (42))  // => 42
fromEither (right (42)) // => 42

fold :: (b -> a -> b) -> b -> Either l a -> b

Reduces a Right with a curried binary function and initial; returns initial for Left.

fold (acc => x => acc + x) (0) (right (5)) // => 5
fold (acc => x => acc + x) (0) (left ('x')) // => 0

lefts :: Array (Either a b) -> Array a

Filters to Left values and extracts them.

lefts ([left (1), right (2), left (3)]) // => [1, 3]

rights :: Array (Either a b) -> Array b

Filters to Right values and extracts them.

rights ([left (1), right (2), right (3)]) // => [2, 3]

partition :: Array (Either a b) -> [Array b, Array a]

Splits an array into [rights, lefts], extracting the inner values.

partition ([right (1), left ('e'), right (2)]) // => [[1, 2], ['e']]

encase :: (a -> b) -> a -> Either Error b

Lifts a throwing unary function into a total Either-returning one.
The error is placed in Left as-is (typically an Error object).
For custom error mapping use `tryCatch`.

encase (JSON.parse) ('{"a":1}') // => right({ a: 1 })
encase (JSON.parse) ('bad')     // => left(SyntaxError)

toMaybe :: Either a b -> Maybe b

Converts Right to Just and Left to Nothing.

toMaybe (right (1))   // => just(1)
toMaybe (left ('err')) // => nothing()

swap :: Either a b -> Either b a

Swaps Left and Right.

swap (left (1))  // => right(1)
swap (right (1)) // => left(1)

map :: (b -> c) -> Either a b -> Either a c

Maps over the Right value, leaving Left untouched.

map (x => x + 1) (right (1))   // => right(2)
map (x => x + 1) (left ('err')) // => left('err')

mapLeft :: (a -> c) -> Either a b -> Either c b

Maps over the Left value, leaving Right untouched.

mapLeft (x => x + '!') (left ('err'))  // => left('err!')
mapLeft (x => x + '!') (right (1))     // => right(1)

bimap :: (a -> c) -> (b -> d) -> Either a b -> Either c d

Maps Left with the first function and Right with the second. Fully curried.

bimap (l => l + '!') (r => r + 1) (right (2)) // => right(3)
bimap (l => l + '!') (r => r + 1) (left ('x')) // => left('x!')

ap :: Either a (b -> c) -> Either a b -> Either a c

Applies a function in a Right to a value in a Right.

ap (right (x => x + 1)) (right (2))   // => right(3)
ap (left ('err')) (right (2))          // => left('err')

chain :: (b -> Either a c) -> Either a b -> Either a c

Monadic bind over Right (chain).

chain (x => right (x + 1)) (right (2))   // => right(3)
chain (x => right (x + 1)) (left ('err')) // => left('err')

chainLeft :: (a -> Either c b) -> Either a b -> Either c b

Monadic bind over Left.

chainLeft (e => left (e + '!')) (left ('x'))  // => left('x!')
chainLeft (e => left (e + '!')) (right (1))   // => right(1)

chainFirst :: (b -> Either a c) -> Either a b -> Either a b

Runs f on the Right value for a potential short-circuit Left;
returns the original Right if f returns Right.

chainFirst (x => left ('stop')) (right (1)) // => left('stop')
chainFirst (x => right ('ok'))  (right (1)) // => right(1)
chainFirst (x => left ('stop')) (left ('e')) // => left('e')

alt :: Either a b -> Either a b -> Either a b

Returns the first Right, or the second if both are Left.

alt (right (2)) (left ('x'))  // => right(2)
alt (left ('y')) (right (1))  // => right(1)
alt (right (2)) (right (1))   // => right(1)

traverse :: (b -> f b) -> ((a -> b) -> f a -> f b) -> (a -> f b) -> Either l a -> f (Either l b)

Applicative traversal over the Right value.

traverse (Array.of) (f => xs => xs.map (f)) (x => [x, x]) (right (1)) // => [right(1), right(1)]

Validation

Applicative error accumulation — Failure | Success.

failure :: e -> Validation e a

Wraps errors in the Failure constructor.
By convention errors is an Array, but any Semigroup is valid.

failure (['too short']) // => { tag: 'failure', failure: ['too short'] }

success :: a -> Validation e a

Wraps a value in the Success constructor.

success (42) // => { tag: 'success', value: 42 }

isFailure :: a -> Boolean

True when the value is a Failure.

isFailure (failure (['err'])) // => true
isFailure (success (1))       // => false

isSuccess :: a -> Boolean

True when the value is a Success.

isSuccess (success (1))       // => true
isSuccess (failure (['err'])) // => false

isValidation :: a -> Boolean

True when the value is a Failure or Success.

isValidation (success (1)) // => true
isValidation (42)          // => false

fromPredicate :: (a -> Boolean) -> e -> a -> Validation e a

Returns Success(a) when the predicate holds, Failure(e) otherwise.

fromPredicate (x => x > 0) (['must be positive']) (-1) // => failure(['must be positive'])
fromPredicate (x => x > 0) (['must be positive']) (3)  // => success(3)

fromNullable :: e -> a -> Validation e a

Returns Success(a) for non-null/undefined, Failure(e) otherwise.

fromNullable (['required']) (null) // => failure(['required'])
fromNullable (['required']) (42)   // => success(42)

validation :: (e -> b) -> (a -> b) -> Validation e a -> b

Case analysis — applies onFailure to the errors or onSuccess to the value.

validation (es => es.join (', ')) (x => x) (success (42))          // => 42
validation (es => es.join (', ')) (x => x) (failure (['oops']))    // => 'oops'

map :: (a -> b) -> Validation e a -> Validation e b

Applies f to the value inside Success; passes Failure through unchanged.

map (x => x + 1) (success (1))        // => success(2)
map (x => x + 1) (failure (['err']))  // => failure(['err'])

mapFailure :: (e -> f) -> Validation e a -> Validation f a

Applies f to the errors inside Failure; passes Success through unchanged.

mapFailure (es => es.map (e => `Error: ${e}`)) (failure (['oops'])) // => failure(['Error: oops'])
mapFailure (es => es.map (e => `Error: ${e}`)) (success (1))        // => success(1)

bimap :: (e -> f) -> (a -> b) -> Validation e a -> Validation f b

Maps Success with fr and Failure with fl.

bimap (es => es.length) (x => x * 2) (success (3))         // => success(6)
bimap (es => es.length) (x => x * 2) (failure (['a','b'])) // => failure(2)

of :: a -> Validation e a

Lifts a value into Validation as a Success.

of (42) // => success(42)

ap :: (e -> e -> e) -> Validation e (a -> b) -> Validation e a -> Validation e b

Applicative ap — accumulates errors from both sides using `concat`.
When both are Failure the errors are combined with the provided semigroup
concat. When both are Success the function is applied.

This is the key difference from Either: both sides are always evaluated.

const apArr = ap (a => b => a.concat (b))
apArr (success (x => x + 1))        (success (2))         // => success(3)
apArr (failure (['bad fn']))         (success (2))         // => failure(['bad fn'])
apArr (success (x => x + 1))        (failure (['bad val']))// => failure(['bad val'])
apArr (failure (['bad fn']))         (failure (['bad val']))// => failure(['bad fn','bad val'])

alt :: Validation e a -> Validation e a -> Validation e a

Returns the first Success, or the second argument if both are Failure.
Errors are NOT accumulated here — use `ap` for that.

alt (success (2))       (failure (['e'])) // => success(2)
alt (failure (['e']))   (success (1))     // => success(1)
alt (failure (['a']))   (failure (['b'])) // => failure(['b'])

fold :: (b -> a -> b) -> b -> Validation e a -> b

Reduces a Success with f and init; returns init for Failure.

fold (acc => x => acc + x) (0) (success (5))       // => 5
fold (acc => x => acc + x) (0) (failure (['err'])) // => 0

traverse :: (b -> f b) -> ((a -> b) -> f a -> f b) -> (a -> f b) -> Validation e a -> f (Validation e b)

Sequences a Success through an applicative functor; Failure is lifted as-is.
Accepts explicit apOf / apMap to remain functor-agnostic.

traverse (Array.of) (f => xs => xs.map (f)) (x => [x, -x]) (success (1))
// => [success(1), success(-1)]
traverse (Array.of) (f => xs => xs.map (f)) (x => [x, -x]) (failure (['e']))
// => [failure(['e'])]

toMaybe :: Validation e a -> Maybe a

Converts Success to Just and Failure to Nothing.

toMaybe (success (1))      // => just(1)
toMaybe (failure (['err'])) // => nothing()

toEither :: Validation e a -> Either e a

Converts Success to Right and Failure to Left.

toEither (success (1))       // => right(1)
toEither (failure (['err'])) // => left(['err'])

fromEither :: Either e a -> Validation e a

Converts Right to Success and Left to Failure.

fromEither (right (1))     // => success(1)
fromEither (left (['err'])) // => failure(['err'])

getFailure :: Validation (Array e) a -> Array e

Extracts errors from a Failure, or returns an empty array for Success.

getFailure (failure (['a', 'b'])) // => ['a', 'b']
getFailure (success (1))          // => []

failures :: Array (Validation (Array e) a) -> Array e

Collects all Failure errors from an array of Validations into a flat array.

failures ([success (1), failure (['a']), failure (['b'])]) // => ['a', 'b']

successes :: Array (Validation e a) -> Array a

Collects the values from all Success elements of an array.

successes ([success (1), failure (['e']), success (2)]) // => [1, 2]

partition :: Array (Validation e a) -> [Array a, Array e]

Splits an array of Validations into [successes, failures],
extracting the inner values from each.

partition ([success (1), failure (['e']), success (2)])
// => [[1, 2], ['e']]

These

The These type — This | That | Both. Holds one or both values.

this_ :: a -> These a b

Constructs a This — holds only the left (a) value.

this_ ('warn') // => { tag: 'this', this: 'warn' }

that :: b -> These a b

Constructs a That — holds only the right (b) value.

that (42) // => { tag: 'that', that: 42 }

both :: a -> b -> These a b

Constructs a Both — holds both values simultaneously.

both ('warn') (42) // => { tag: 'both', this: 'warn', that: 42 }

isThis :: a -> Boolean

True when the value is a This.

isThis (this_ ('x')) // => true
isThis (that (1))    // => false

isThat :: a -> Boolean

True when the value is a That.

isThat (that (1))    // => true
isThat (this_ ('x')) // => false

isBoth :: a -> Boolean

True when the value is a Both.

isBoth (both ('w') (1)) // => true
isBoth (that (1))       // => false

isThese :: a -> Boolean

True when the value is a This, That, or Both.

isThese (that (1))    // => true
isThese (42)          // => false

hasThis :: These a b -> Boolean

True when the These contains a left (a) value — This or Both.

hasThis (this_ ('w'))    // => true
hasThis (both ('w') (1)) // => true
hasThis (that (1))       // => false

hasThat :: These a b -> Boolean

True when the These contains a right (b) value — That or Both.

hasThat (that (1))       // => true
hasThat (both ('w') (1)) // => true
hasThat (this_ ('w'))    // => false

these :: (a -> c) -> (b -> c) -> (a -> b -> c) -> These a b -> c

Full case analysis on These.

these (a => `warn:${a}`) (b => b * 2) (a => b => b) (that (21))       // => 42
these (a => `warn:${a}`) (b => b * 2) (a => b => b) (this_ ('oops'))  // => 'warn:oops'
these (a => `warn:${a}`) (b => b * 2) (a => b => b) (both ('w') (21)) // => 21

getThis :: These a b -> Maybe a

Returns Just the left (a) value if present, Nothing otherwise.

getThis (this_ ('w'))    // => just('w')
getThis (both ('w') (1)) // => just('w')
getThis (that (1))       // => nothing()

getThat :: These a b -> Maybe b

Returns Just the right (b) value if present, Nothing otherwise.

getThat (that (1))       // => just(1)
getThat (both ('w') (1)) // => just(1)
getThat (this_ ('w'))    // => nothing()

fromThis :: a -> These a b -> a

Extracts the left value, returning the default for That.

fromThis ('def') (this_ ('w'))    // => 'w'
fromThis ('def') (both ('w') (1)) // => 'w'
fromThis ('def') (that (1))       // => 'def'

fromThat :: b -> These a b -> b

Extracts the right value, returning the default for This.

fromThat (0) (that (1))       // => 1
fromThat (0) (both ('w') (1)) // => 1
fromThat (0) (this_ ('w'))    // => 0

equals :: (a -> a -> Boolean) -> (b -> b -> Boolean) -> These a b -> These a b -> Boolean

Structural equality using separate comparators for each side.

equals (a => b => a === b) (a => b => a === b) (that (1)) (that (1))          // => true
equals (a => b => a === b) (a => b => a === b) (both ('w') (1)) (both ('w') (1)) // => true
equals (a => b => a === b) (a => b => a === b) (this_ ('w')) (that (1))       // => false

concat :: (a -> a -> a) -> (b -> b -> b) -> These a b -> These a b -> These a b

Combines two These values, accumulating both sides with their respective
Semigroup concats. The table (using Haskell notation):

This a <> This b = This (a <> b)
This a <> That y = Both a y
This a <> Both b y = Both (a <> b) y
That x <> This b = Both b x
That x <> That y = That (x <> y)
That x <> Both b y = Both b (x <> y)
Both a x <> This b = Both (a <> b) x
Both a x <> That y = Both a (x <> y)
Both a x <> Both b y = Both (a <> b) (x <> y)

const c = concat (a => b => a + b) (a => b => a + b)
c (this_ ('a')) (this_ ('b'))     // => this_('ab')
c (that (1))    (that (2))        // => that(3)
c (this_ ('a')) (that (1))        // => both('a')(1)
c (both ('a') (1)) (both ('b') (2)) // => both('ab')(3)

map :: (b -> c) -> These a b -> These a c

Applies f to the right (b) value; passes This through unchanged.

map (x => x + 1) (that (1))        // => that(2)
map (x => x + 1) (both ('w') (1))  // => both('w')(2)
map (x => x + 1) (this_ ('w'))     // => this_('w')

mapThis :: (a -> c) -> These a b -> These c b

Applies f to the left (a) value; passes That through unchanged.

mapThis (x => x + '!') (this_ ('w'))    // => this_('w!')
mapThis (x => x + '!') (both ('w') (1)) // => both('w!')(1)
mapThis (x => x + '!') (that (1))       // => that(1)

bimap :: (a -> c) -> (b -> d) -> These a b -> These c d

Maps both sides independently.

bimap (a => a + '!') (b => b * 2) (both ('w') (3)) // => both('w!')(6)
bimap (a => a + '!') (b => b * 2) (that (3))       // => that(6)
bimap (a => a + '!') (b => b * 2) (this_ ('w'))    // => this_('w!')

of :: b -> These a b

Lifts a value into These as a That.

of (42) // => that(42)

ap :: (a -> a -> a) -> These a (b -> c) -> These a b -> These a c

Apply — combines These values, accumulating the left side with concatA.
This is analogous to Validation's ap but operates on These.

const apStr = ap (a => b => a + b)
apStr (that (x => x + 1))        (that (2))         // => that(3)
apStr (this_ ('e1'))              (that (2))         // => this_('e1')
apStr (that (x => x + 1))        (this_ ('e2'))      // => this_('e2')
apStr (this_ ('e1'))              (this_ ('e2'))      // => this_('e1e2')
apStr (both ('w') (x => x + 1))  (that (2))         // => both('w')(3)
apStr (both ('w1') (x => x + 1)) (both ('w2') (2)) // => both('w1w2')(3)

chain :: (a -> a -> a) -> (b -> These a c) -> These a b -> These a c

Monadic bind — threads the left side through using concatA.

const chainStr = chain (a => b => a + b)
chainStr (x => that (x + 1))       (that (2))         // => that(3)
chainStr (x => both ('w2') (x + 1)) (both ('w1') (2)) // => both('w1w2')(3)
chainStr (x => that (x + 1))        (this_ ('e'))      // => this_('e')

fold :: (b -> a -> b) -> b -> These e a -> b

Reduces the right (b) value with f and init; returns init for This.

fold (acc => x => acc + x) (0) (that (5))        // => 5
fold (acc => x => acc + x) (0) (both ('w') (5))  // => 5
fold (acc => x => acc + x) (0) (this_ ('w'))     // => 0

toMaybe :: These a b -> Maybe b

Converts to Maybe — Just for That/Both (the right value), Nothing for This.

toMaybe (that (1))       // => just(1)
toMaybe (both ('w') (1)) // => just(1)
toMaybe (this_ ('w'))    // => nothing()

toEither :: These a b -> Either a b

Converts to Either — Right for That/Both, Left for This.

toEither (that (1))       // => right(1)
toEither (both ('w') (1)) // => right(1)
toEither (this_ ('w'))    // => left('w')

swap :: These a b -> These b a

Swaps This and That — both sides exchange roles.

swap (this_ ('w'))    // => that('w')
swap (that (1))       // => this_(1)
swap (both ('w') (1)) // => both(1)('w')

Ordering

The Ordering type — LT | EQ | GT. Comparator algebra.

lt :: Ordering

Less-than ordering.

lt // => { tag: 'LT' }

eq :: Ordering

Equal ordering — the identity element of the Ordering Monoid.

eq // => { tag: 'EQ' }

gt :: Ordering

Greater-than ordering.

gt // => { tag: 'GT' }

isOrdering :: a -> Boolean

True when the value is an Ordering.

isOrdering (lt) // => true
isOrdering (0)  // => false

isLT :: Ordering -> Boolean

True when the Ordering is LT.

isLT (lt) // => true
isLT (eq) // => false

isEQ :: Ordering -> Boolean

True when the Ordering is EQ.

isEQ (eq) // => true
isEQ (lt) // => false

isGT :: Ordering -> Boolean

True when the Ordering is GT.

isGT (gt) // => true
isGT (eq) // => false

equals :: Ordering -> Ordering -> Boolean

True when both Orderings are the same constructor.

equals (lt) (lt) // => true
equals (lt) (gt) // => false

concat :: Ordering -> Ordering -> Ordering

Monoid concat — returns the first non-EQ value, or EQ if both are EQ.
This is the key operation for chaining multi-field comparators:
the first comparison that is not EQ determines the result.

concat (lt) (gt) // => lt
concat (eq) (gt) // => gt
concat (eq) (eq) // => eq

empty :: Ordering

The identity element for concat — EQ.

empty // => { tag: 'EQ' }

ordering :: a -> a -> a -> Ordering -> a

Case analysis on an Ordering.

ordering (-1) (0) (1) (lt) // => -1
ordering (-1) (0) (1) (eq) // => 0
ordering (-1) (0) (1) (gt) // => 1

toNumber :: Ordering -> -1 | 0 | 1

Converts an Ordering to the conventional JS comparator integer.

toNumber (lt) // => -1
toNumber (eq) // => 0
toNumber (gt) // => 1

fromNumber :: Number -> Ordering

Converts a JS comparator integer (negative / zero / positive) to Ordering.

fromNumber (-3) // => lt
fromNumber (0)  // => eq
fromNumber (7)  // => gt

compare :: a -> a -> Ordering

Derives an Ordering from a value using its natural JS ordering.
Equivalent to Haskell's `compare` for types with (<) and (>).

compare (1) (2) // => lt
compare (2) (2) // => eq
compare (3) (2) // => gt

comparing :: Ord b => (a -> b) -> a -> a -> Ordering

Builds a comparator by projecting both values through f before comparing.
The primary tool for building sort keys — equivalent to Haskell's
`comparing`.

comparing (x => x.age) ({ age: 30 }) ({ age: 25 }) // => gt

invert :: Ordering -> Ordering

Reverses an Ordering — swaps LT and GT, keeps EQ.
Use with `comparing` to sort descending.

invert (lt) // => gt
invert (eq) // => eq
invert (gt) // => lt

toComparator :: (a -> a -> Ordering) -> (a, a) -> Number

Converts a curried Ordering comparator to a JS-style comparator
(a, b) => number, suitable for use with Array.prototype.sort directly.

[3, 1, 2].sort (toComparator (compare)) // => [1, 2, 3]

sortWith :: (a -> a -> Ordering) -> Array a -> Array a

Stable sort of an array using an Ordering comparator.
Replaces the boolean-lte sort from array.js with a richer interface.

sortWith (compare) ([3, 1, 2]) // => [1, 2, 3]
sortWith (comparing (x => x.n)) ([{ n: 2 }, { n: 1 }]) // => [{ n: 1 }, { n: 2 }]

concatComparators :: (a -> a -> Ordering) -> (a -> a -> Ordering) -> a -> a -> Ordering

Combines two Ordering comparators — the second is used only when the
first yields EQ. This is just `concat` lifted to comparator functions,
enabling clean multi-field sorts.

const byAge  = comparing (x => x.age)
const byName = comparing (x => x.name)
sortWith (concatComparators (byAge) (byName)) (people)

Identity

The Identity functor / monad. The simplest possible wrapper.

identity :: a -> Identity a

Wraps a value in the Identity constructor.

identity (42) // => { tag: 'identity', value: 42 }

isIdentity :: a -> Boolean

True when the value is an Identity.

isIdentity (identity (1)) // => true
isIdentity (42)           // => false

extract :: Identity a -> a

Extracts the wrapped value.

extract (identity (42)) // => 42

equals :: (a -> a -> Boolean) -> Identity a -> Identity a -> Boolean

True when both Identity values are equal under the given comparator.

equals (a => b => a === b) (identity (1)) (identity (1)) // => true
equals (a => b => a === b) (identity (1)) (identity (2)) // => false

lte :: (a -> a -> Boolean) -> Identity a -> Identity a -> Boolean

Ordering of two Identity values under the given lte comparator.

lte (a => b => a <= b) (identity (1)) (identity (2)) // => true
lte (a => b => a <= b) (identity (2)) (identity (1)) // => false

map :: (a -> b) -> Identity a -> Identity b

Applies f to the wrapped value and re-wraps the result.

map (x => x + 1) (identity (1)) // => identity(2)

of :: a -> Identity a

Lifts a value into Identity. Alias for `identity`.

of (42) // => identity(42)

ap :: Identity (a -> b) -> Identity a -> Identity b

Applies the function inside the first Identity to the value in the second.

ap (identity (x => x + 1)) (identity (2)) // => identity(3)

chain :: (a -> Identity b) -> Identity a -> Identity b

Monadic bind — applies f to the wrapped value.

chain (x => identity (x + 1)) (identity (2)) // => identity(3)

fold :: (b -> a -> b) -> b -> Identity a -> b

Reduces the single wrapped value with f and init.

fold (acc => x => acc + x) (0) (identity (5)) // => 5

traverse :: (b -> f b) -> ((a -> b) -> f a -> f b) -> (a -> f b) -> Identity a -> f (Identity b)

Sequences the Identity through an applicative functor.
Accepts explicit apOf / apMap to remain functor-agnostic.

apOf :: b -> f b (pure / of)
apMap :: (a -> b) -> f a -> f b (map, curried)

const apOf  = Array.of
const apMap = f => xs => xs.map (f)
traverse (apOf) (apMap) (x => [x, -x]) (identity (1)) // => [identity(1), identity(-1)]

extend :: (Identity a -> b) -> Identity a -> Identity b

Comonad extend — wraps the whole Identity in another Identity via f.

extend (fa => fa.value + 1) (identity (2)) // => identity(3)

State

The State monad. Stateful computations as pure functions.

state :: (s -> [a, s]) -> State s a

Wraps a state-transition function into the State monad.
The function must have type s -> [a, s].

state (s => [s + 1, s + 1]) // a State action that increments and returns

get :: State s s

Returns the current state as the value, leaving it unchanged.
Haskell: `get`

run (get) (42) // => [42, 42]

put :: s -> State s null

Replaces the state with a new value, returning unit (null).
Haskell: `put`

run (put (99)) (0) // => [null, 99]

modify :: (s -> s) -> State s null

Applies a function to the state, replacing it with the result.
Returns unit (null).
Haskell: `modify`

run (modify (s => s + 1)) (0) // => [null, 1]

gets :: (s -> a) -> State s a

Projects a value from the current state without changing it.
Haskell: `gets`

run (gets (s => s * 2)) (21) // => [42, 21]

run :: State s a -> s -> [a, s]

Runs a State action with an initial state, returning [value, finalState].

run (of (42)) (0) // => [42, 0]

eval_ :: State s a -> s -> a

Runs a State action and returns only the final value.

eval (of (42)) (0) // => 42

exec :: State s a -> s -> s

Runs a State action and returns only the final state.

exec (put (99)) (0) // => 99

map :: (a -> b) -> State s a -> State s b

Applies f to the value produced by a State action, leaving the state
transition unchanged.

run (map (x => x * 2) (of (21))) (0) // => [42, 0]

of :: a -> State s a

Lifts a pure value into State — the state passes through unchanged.

run (of (42)) (99) // => [42, 99]

ap :: State s (a -> b) -> State s a -> State s b

Sequential application — runs the function action, then the value action,
threading the state through both.

run (ap (of (x => x + 1)) (of (41))) (0) // => [42, 0]

chain :: (a -> State s b) -> State s a -> State s b

Sequences two State actions — the second action can depend on the value
produced by the first. The state threads through both.

const action = chain (n => put (n)) (gets (s => s + 1))
run (action) (0) // => [null, 1]

andThen :: State s b -> State s a -> State s b

Sequences two State actions, discarding the value of the first.
Haskell: (>>)

run (andThen (of (42)) (put (99))) (0) // => [42, 99]

chainRec :: ((a -> Step, b -> Step, a) -> State s Step) -> a -> State s b

Stack-safe tail-recursive State bind (Fantasy Land ChainRec).
f receives (next, done, a) and must return a State action that yields a Step.

// Count down from n, accumulating increments in state:
const action = chainRec ((next, done, n) =>
  n <= 0 ? of (done (0)) : chain ((_) => of (next (n - 1))) (modify (s => s + 1))
) (5)
run (action) (0) // => [0, 5]

sequence :: Array (State s a) -> State s (Array a)

Runs a list of State actions in sequence, collecting all values.
The state threads through each action left to right.

run (sequence ([of (1), of (2), of (3)])) (0) // => [[1, 2, 3], 0]

traverse :: (a -> State s b) -> Array a -> State s (Array b)

Maps f over an array, running each resulting State action in sequence and
collecting the values. Equivalent to sequence (xs.map (f)).

run (traverse (x => of (x * 2)) ([1, 2, 3])) (0) // => [[2, 4, 6], 0]

lift2 :: (a -> b -> c) -> State s a -> State s b -> State s c

Runs two State actions and combines their values with a curried binary
function. Useful for point-free style when you don't need intermediate
binds.

run (lift2 (a => b => a + b) (of (1)) (of (2))) (0) // => [3, 0]

StateEither

The StateEither monad. Stateful computations with error handling.

state :: (s -> Either e [a, s]) -> StateT s e a

Wraps a state-transition function into the StateT monad.
The function must have type s -> Either e [a, s].

state (s => E.right ([s + 1, s + 1]))

get :: StateT s e s

Returns the current state as the value, leaving it unchanged.

run (get) (42) // => right([42, 42])

put :: s -> StateT s e null

Replaces the state with a new value, returning unit (null).

run (put (99)) (0) // => right([null, 99])

modify :: (s -> s) -> StateT s e null

Applies a function to the state, replacing it with the result.
Returns unit (null).

run (modify (s => s + 1)) (0) // => right([null, 1])

gets :: (s -> a) -> StateT s e a

Projects a value from the current state without changing it.

run (gets (s => s * 2)) (21) // => right([42, 21])

lift :: Either e a -> StateT s e a

Lifts an Either value into StateT — the state passes through unchanged.
Use this to inject a pre-computed Either result into a StateT pipeline.

run (lift (E.right (42))) (0)       // => right([42, 0])
run (lift (E.left ('oops'))) (0)    // => left('oops')

run :: StateT s e a -> s -> Either e [a, s]

Runs a StateT action with an initial state, returning Either e [value, finalState].

run (of (42)) (0) // => right([42, 0])

eval_ :: StateT s e a -> s -> Either e a

Runs a StateT action and returns only the final value inside Either.

eval_ (of (42)) (0) // => right(42)

exec :: StateT s e a -> s -> Either e s

Runs a StateT action and returns only the final state inside Either.

exec (put (99)) (0) // => right(99)

map :: (a -> b) -> StateT s e a -> StateT s e b

Applies f to the value produced by a StateT action, leaving the state
transition unchanged. Short-circuits on Left.

run (map (x => x * 2) (of (21))) (0) // => right([42, 0])

of :: a -> StateT s e a

Lifts a pure value into StateT — the state passes through unchanged.

run (of (42)) (99) // => right([42, 99])

ap :: StateT s e (a -> b) -> StateT s e a -> StateT s e b

Sequential application — runs the function action, then the value action,
threading the state through both. Short-circuits on the first Left.

run (ap (of (x => x + 1)) (of (41))) (0) // => right([42, 0])

chain :: (a -> StateT s e b) -> StateT s e a -> StateT s e b

Sequences two StateT actions — the second can depend on the value of the
first. Short-circuits on the first Left.

const action = chain (n => put (n)) (gets (s => s + 1))
run (action) (0) // => right([null, 1])

andThen :: StateT s e b -> StateT s e a -> StateT s e b

Sequences two StateT actions, discarding the value of the first.
Haskell: (>>)

run (andThen (of (42)) (put (99))) (0) // => right([42, 99])

chainFirst :: (a -> StateT s e b) -> StateT s e a -> StateT s e a

Runs f on the Right value for a potential short-circuit Left;
returns the original action's result unchanged if f succeeds.
Mirrors Either's chainFirst — useful for validation side-checks.

run (chainFirst (x => lift (E.left ('stop'))) (of (1))) (0) // => left('stop')
run (chainFirst (x => of ('ok')) (of (1))) (0)              // => right([1, 0])

chainRec :: ((a -> Step, b -> Step, a) -> StateT s e Step) -> a -> StateT s e b

Stack-safe tail-recursive StateT bind (Fantasy Land ChainRec).

const count = chainRec ((next, done, n) =>
  n <= 0 ? of (done (0)) : andThen (of (next (n - 1))) (modify (s => s + 1))
) (5)
run (count) (0) // => right([0, 5])

sequence :: Array (StateT s e a) -> StateT s e (Array a)

Runs a list of StateT actions in sequence, collecting all values.
Short-circuits on the first Left.

run (sequence ([of (1), of (2), of (3)])) (0) // => right([[1, 2, 3], 0])

traverse :: (a -> StateT s e b) -> Array a -> StateT s e (Array b)

Maps f over an array, running each resulting StateT action in sequence and
collecting the values. Short-circuits on the first Left.
Equivalent to sequence (xs.map (f)).

run (traverse (x => of (x * 2)) ([1, 2, 3])) (0) // => right([[2, 4, 6], 0])

lift2 :: (a -> b -> c) -> StateT s e a -> StateT s e b -> StateT s e c

Runs two StateT actions and combines their values with a curried binary
function. Short-circuits on the first Left.

run (lift2 (a => b => a + b) (of (1)) (of (2))) (0) // => right([3, 0])

Lens

Functional lenses for immutable data access and update.

lens :: (s -> a) -> (a -> s -> s) -> Lens s a

Constructs a Lens from a getter and a curried setter.

const nameLens = lens (p => p.name) (v => p => ({ ...p, name: v }))
view (nameLens) ({ name: 'Alice', age: 30 }) // => 'Alice'

prop :: String -> Lens Object a

Focuses on the value at a given key in a plain object.
`set` returns a shallow copy with the key updated.

view (prop ('name')) ({ name: 'Alice' }) // => 'Alice'
over (prop ('age')) (x => x + 1) ({ name: 'Alice', age: 30 }) // => { name: 'Alice', age: 31 }

index :: Integer -> Lens (Array a) (Maybe a)

Focuses on the element at a given array index.
`get` returns Just(element) when in bounds, Nothing otherwise.
`set` returns the array with the element replaced, or the original array
if the index is out of bounds.

view (index (1)) ([10, 20, 30]) // => just(20)
view (index (9)) ([10, 20, 30]) // => nothing()
over (index (1)) (M.map (x => x * 2)) ([10, 20, 30]) // => [10, 40, 30]

view :: Lens s a -> s -> a

Reads the focused value out of a structure.

view (prop ('x')) ({ x: 42 }) // => 42

set :: Lens s a -> a -> s -> s

Replaces the focused value in a structure.

set (prop ('x')) (99) ({ x: 42, y: 1 }) // => { x: 99, y: 1 }

over :: Lens s a -> (a -> a) -> s -> s

Applies a function to the focused value, returning an updated structure.

over (prop ('x')) (x => x + 1) ({ x: 42 }) // => { x: 43 }

composeLens :: Lens s a -> Lens a b -> Lens s b

Composes two lenses left-to-right: the first lens focuses on a part of `s`,
the second focuses on a part of that part.

compose (prop ('address')) (prop ('city'))
focuses on the `city` field inside the `address` field of a structure.

const cityLens = composeLens (prop ('address')) (prop ('city'))
view (cityLens) ({ address: { city: 'Oslo' } })           // => 'Oslo'
over (cityLens) (s => s.toUpperCase ()) ({ address: { city: 'Oslo' } })
// => { address: { city: 'OSLO' } }

overWithResult :: Lens s a -> (a -> [b, a]) -> s -> [b, s]

Applies a function to the focused value and returns both the result and
the updated structure as a pair [result, newStructure].
Useful when the transform function produces a value you want to inspect.

overWithResult (prop ('n')) (x => [x * 2, x + 1]) ({ n: 5 })
// => [10, { n: 6 }]

overWhen :: Lens s a -> (a -> Boolean) -> (a -> a) -> s -> s

Modifies the focused value only when the predicate holds.
Returns the structure unchanged if the predicate fails.

overWhen (prop ('age')) (x => x >= 18) (x => x + 1) ({ age: 20 }) // => { age: 21 }
overWhen (prop ('age')) (x => x >= 18) (x => x + 1) ({ age: 10 }) // => { age: 10 }

toListOf :: Lens s a -> Array s -> Array a

Collects the focused value from each element of an array.

toListOf (prop ('name')) ([{ name: 'Alice' }, { name: 'Bob' }])
// => ['Alice', 'Bob']

≡ sequence through xs applying f to each element

// => nothing()

Ordered pair (2-tuple) utilities.

pair :: a -> b -> [a, b]

Constructs a 2-element tuple.

pair (1) (2) // => [1, 2]

of :: a -> [a, a]

Lifts a value into a pair context — pairs it with itself.

of (3) // => [3, 3]

fst :: [a, b] -> a

Extracts the first element.

fst ([1, 2]) // => 1

snd :: [a, b] -> b

Extracts the second element.

snd ([, b]) // => b
snd ([1, 2]) // => 2

fold :: (a -> b -> c) -> [a, b] -> c

Applies a curried binary function to the pair's elements.
The canonical eliminator — Haskell's `uncurry`.

fold (a => b => a + b) ([1, 2]) // => 3

map :: (b -> d) -> [a, b] -> [a, d]

Maps over the second element (the Functor slot), leaving the first unchanged.
Consistent with the standard Functor instance for pairs, where the first
element is the "fixed" context and the second is the mapped value.

map (x => x + 1) ([1, 2]) // => [1, 3]

mapFst :: (a -> c) -> [a, b] -> [c, b]

Maps over the first element, leaving the second unchanged.

mapFst (x => x + 1) ([1, 2]) // => [2, 2]

mapSnd :: (b -> d) -> [a, b] -> [a, d]

Maps over the second element, leaving the first unchanged.
Alias for `map` — provided for symmetry with `mapFst`.

mapSnd (x => x + 1) ([1, 2]) // => [1, 3]

bimap :: (a -> c) -> (b -> d) -> [a, b] -> [c, d]

Maps both elements independently.

bimap (x => x + 1) (x => x * 2) ([1, 3]) // => [2, 6]

ap :: [a -> b, a] -> b

Applies the function in the first slot to the value in the second.
`ap ([f, x])` ≡ `f (x)` — the pair acts as an environment holding
both the function and its argument.

ap ([x => x + 1, 5]) // => 6

swap :: [a, b] -> [b, a]

Swaps the two elements.

swap ([1, 2]) // => [2, 1]

traverse :: (b -> f b) -> ((a -> b) -> f a -> f b) -> (a -> f b) -> [a, c] -> f [b, c]

Traverses the first element through an applicative functor.
Accepts explicit apOf / apMap to remain functor-agnostic.

apOf :: b -> f b (pure / of)
apMap :: (a -> b) -> f a -> f b (map, curried)

const apOf  = Array.of
const apMap = f => xs => xs.map (f)
traverse (apOf) (apMap) (x => [x, -x]) ([1, 2]) // => [[1, 2], [-1, 2]]

StrMap

Plain JS objects used as string-keyed maps.

equals :: (v -> v -> Boolean) -> StrMap v -> StrMap v -> Boolean

Element-wise equality over sorted keys.

equals (a => b => a === b) ({ a: 1 }) ({ a: 1 }) // => true
equals (a => b => a === b) ({ a: 1 }) ({ a: 2 }) // => false

lte :: (v -> v -> Boolean) -> StrMap v -> StrMap v -> Boolean

Lexicographic ordering over sorted keys; values compared on key ties.

lte (a => b => a <= b) ({ a: 1 }) ({ a: 2 }) // => true
lte (a => b => a <= b) ({ a: 2 }) ({ a: 1 }) // => false

concat :: StrMap v -> StrMap v -> StrMap v

Right-biased merge — keys in b overwrite keys in a.

concat ({ a: 1 }) ({ b: 2 })       // => { a: 1, b: 2 }
concat ({ a: 1 }) ({ a: 2, b: 3 }) // => { a: 2, b: 3 }

empty :: () -> StrMap v

Returns a new empty object — the identity element for concat.

empty () // => {}

singleton :: String -> a -> StrMap a

Creates a single-entry map.

singleton ('a') (1) // => { a: 1 }

fromPairs :: Array [String, a] -> StrMap a

Builds a map from [key, value] pairs — last write wins on duplicate keys.

fromPairs ([['a', 1], ['b', 2]]) // => { a: 1, b: 2 }

size :: StrMap v -> Integer

Returns the number of keys.

size ({ a: 1, b: 2 }) // => 2
size ({})              // => 0

all :: (v -> Boolean) -> StrMap v -> Boolean

True when every value satisfies the predicate.

all (v => v > 0) ({ a: 1, b: 2 }) // => true
all (v => v > 1) ({ a: 1, b: 2 }) // => false

any :: (v -> Boolean) -> StrMap v -> Boolean

True when at least one value satisfies the predicate.

any (v => v > 1) ({ a: 1, b: 2 }) // => true
any (v => v > 5) ({ a: 1, b: 2 }) // => false

none :: (v -> Boolean) -> StrMap v -> Boolean

True when no value satisfies the predicate.

none (v => v > 5) ({ a: 1, b: 2 }) // => true
none (v => v > 1) ({ a: 1, b: 2 }) // => false

elem :: (v -> v -> Boolean) -> v -> StrMap v -> Boolean

True when x is found among the values using the comparator.

elem (a => b => a === b) (2) ({ a: 1, b: 2 }) // => true
elem (a => b => a === b) (9) ({ a: 1, b: 2 }) // => false

lookup :: String -> StrMap a -> Maybe a

Returns Just(v) if key is an own enumerable property; Nothing otherwise.

lookup ('a') ({ a: 1 }) // => just(1)
lookup ('z') ({ a: 1 }) // => nothing()

insert :: String -> a -> StrMap a -> StrMap a

Returns a new map with the key set to val.

insert ('b') (2) ({ a: 1 }) // => { a: 1, b: 2 }

remove :: String -> StrMap a -> StrMap a

Returns a new map with the key deleted.

remove ('a') ({ a: 1, b: 2 }) // => { b: 2 }

map :: (a -> b) -> StrMap a -> StrMap b

Applies f to every value.

map (v => v * 2) ({ a: 1, b: 2 }) // => { a: 2, b: 4 }

mapWithKey :: (String -> a -> b) -> StrMap a -> StrMap b

Applies f to every key-value pair, passing key then value.

mapWithKey (k => v => `${k}:${v}`) ({ a: 1, b: 2 }) // => { a: 'a:1', b: 'b:2' }

filter :: (v -> Boolean) -> StrMap v -> StrMap v

Keeps only entries whose value satisfies the predicate.

filter (v => v > 1) ({ a: 1, b: 2, c: 3 }) // => { b: 2, c: 3 }

filterWithKey :: (String -> v -> Boolean) -> StrMap v -> StrMap v

Keeps only entries whose key and value satisfy the predicate.

filterWithKey (k => v => k !== 'a' && v > 0) ({ a: 1, b: 2 }) // => { b: 2 }

reject :: (v -> Boolean) -> StrMap v -> StrMap v

Removes entries whose value satisfies the predicate.

reject (v => v > 1) ({ a: 1, b: 2 }) // => { a: 1 }

ap :: StrMap (a -> b) -> StrMap a -> StrMap b

Applies functions to matching keys; only shared keys appear in the result.

ap ({ a: x => x + 1 }) ({ a: 1, b: 2 }) // => { a: 2 }

alt :: StrMap v -> StrMap v -> StrMap v

Left-biased merge — keys from a take priority over keys from b.

alt ({ a: 1 }) ({ a: 9, b: 2 }) // => { a: 1, b: 2 }

reduce :: (b -> a -> b) -> b -> StrMap a -> b

Left fold over values in sorted key order, with a curried binary function.

reduce (acc => v => acc + v) (0) ({ b: 2, a: 1 }) // => 3

foldWithKey :: (b -> String -> a -> b) -> b -> StrMap a -> b

Left fold over key-value pairs in sorted key order.

foldWithKey (acc => k => v => acc + `${k}=${v} `) ('') ({ b: 2, a: 1 })
// => 'a=1 b=2 '

traverse :: (b -> f b) -> (f (a->b) -> f a -> f b) -> ((a->b) -> f a -> f b) -> (v -> f b) -> StrMap v -> f (StrMap b)

Applicative traversal over all values in key-insertion order.

const apOf  = Array.of
const apAp  = ff => fa => ff.flatMap (f => fa.map (f))
const apMap = f  => fa => fa.map (f)
traverse (apOf) (apAp) (apMap) (v => [v, v * 2]) ({ a: 1 })

apOf  :: b -> f b                        (pure / of)
apAp  :: f (a -> b) -> f a -> f b        (ap, curried)
apMap :: (a -> b) -> f a -> f b          (map, curried)

keys :: StrMap a -> Array String

Returns all enumerable keys in insertion order.

keys ({ b: 2, a: 1 }) // => ['b', 'a']

values :: StrMap a -> Array a

Returns all enumerable values in insertion order.

values ({ a: 1, b: 2 }) // => [1, 2]

pairs :: StrMap a -> Array [String, a]

Returns all [key, value] entries in insertion order.

pairs ({ a: 1, b: 2 }) // => [['a', 1], ['b', 2]]

RegExp

RegExp constructors, comparison, and matching utilities.

equals :: RegExp -> RegExp -> Boolean

True when both regexes have the same source and all flags match.

equals (/a/g) (/a/g) // => true

regex :: String -> String -> RegExp

Constructs a RegExp from flags and source.

regex ('g') ('[0-9]+') // => /[0-9]+/g

regexEscape :: String -> String

Escapes all regex metacharacters so the result matches the literal string.

regexEscape ('a.b') // => 'a\.b'

test :: RegExp -> String -> Boolean

Tests whether the pattern matches the string (stateless — resets lastIndex).

test (/^a/) ('abacus') // => true

match :: RegExp -> String -> Maybe (Array String)

Returns Just the array of capture groups if pattern matches, Nothing otherwise.
Unmatched optional groups are returned as empty string ''.

match (/(\w+)\s(\w+)/) ('hello world') // => just(['hello', 'world'])
match (/(\d+)/) ('abc')                // => nothing()

matchAll :: RegExp -> String -> Array (Array String)

Returns all capture-group arrays for all matches (pattern must have the 'g' flag).
Unmatched optional groups are returned as empty string ''.

matchAll (/(\w+)/g) ('hi there') // => [['hi'], ['there']]

replace :: (Array String -> String) -> RegExp -> String -> String

Replaces pattern matches using a substitution function over capture groups.
Unmatched optional groups are passed as ''.

replace (([w]) => w.toUpperCase ()) (/(\w+)/) ('hello world') // => 'HELLO world'
replace (() => 'X') (/a/) ('cat')                             // => 'cXt'

isNil (null)      // => true

isNotNil :: a -> Boolean

Returns true for any value that is not null or undefined.

isNotNil (0)    // => true

fromMaybe (M.just (1))   // => 1

Applies f to the value if non-nil, otherwise propagates null.

Applies f (which may itself return null) to the value if non-nil.

getOrElse :: a -> Nil a -> a

Returns the value if non-nil, otherwise the provided default.

getOrElse (0) (5)    // => 5
getOrElse (0) (null) // => 0

getOrElse_ :: (() -> a) -> Nil a -> a

Returns the value if non-nil, otherwise calls the thunk.

getOrElse_ (() => 0) (null) // => 0

alt :: Nil a -> Nil a -> Nil a

Returns the first non-nil value, or null if both are nil.

alt (5) (null) // => 5
alt (null) (3) // => 3
alt (null) (null) // => null

toMaybe :: Nil a -> Maybe a

Converts a Nil value to Maybe — Just for non-nil, Nothing for nil.

toMaybe (5)    // => just(5)
toMaybe (null) // => nothing()

Map

Functional Map with arbitrary key types. Immutable, equality-function-based.

empty :: () -> Map k v

Returns a new empty Map.

empty () // => []

singleton :: k -> v -> Map k v

Creates a Map with a single key-value pair.

singleton ('a') (1) // => [['a', 1]]

fromPairs :: (k -> k -> Boolean) -> Array [k, v] -> Map k v

Builds a Map from an array of [key, value] pairs using the given equality.
Later entries win on duplicate keys.

fromPairs (a => b => a === b) ([['a', 1], ['b', 2], ['a', 3]])
// => [['a', 3], ['b', 2]]

toPairs :: Map k v -> Array [k, v]

Converts the Map to an array of [key, value] pairs.

toPairs ([['a', 1], ['b', 2]]) // => [['a', 1], ['b', 2]]

size :: Map k v -> Integer

Returns the number of entries in the Map.

size ([['a', 1], ['b', 2]]) // => 2
size ([])                   // => 0

isEmpty :: Map k v -> Boolean

True when the Map has no entries.

isEmpty ([])          // => true
isEmpty ([['a', 1]]) // => false

member :: (k -> k -> Boolean) -> k -> Map k v -> Boolean

True when the Map contains the given key.

member (a => b => a === b) ('a') ([['a', 1]]) // => true
member (a => b => a === b) ('z') ([['a', 1]]) // => false

notMember :: (k -> k -> Boolean) -> k -> Map k v -> Boolean

True when the Map does NOT contain the given key.

notMember (a => b => a === b) ('z') ([['a', 1]]) // => true

lookup :: (k -> k -> Boolean) -> k -> Map k v -> Maybe v

Returns Just(v) if the key is in the Map, Nothing otherwise.

lookup (a => b => a === b) ('a') ([['a', 1]]) // => just(1)
lookup (a => b => a === b) ('z') ([['a', 1]]) // => nothing()

findWithDefault :: (k -> k -> Boolean) -> v -> k -> Map k v -> v

Returns the value for the key, or the provided default if absent.

findWithDefault (a => b => a === b) (0) ('a') ([['a', 1]]) // => 1
findWithDefault (a => b => a === b) (0) ('z') ([['a', 1]]) // => 0

insert :: (k -> k -> Boolean) -> k -> v -> Map k v -> Map k v

Returns a new Map with the key set to val.
If the key already exists its value is replaced.

insert (a => b => a === b) ('b') (2) ([['a', 1]]) // => [['a', 1], ['b', 2]]
insert (a => b => a === b) ('a') (9) ([['a', 1]]) // => [['a', 9]]

insertWith :: (k -> k -> Boolean) -> k -> v -> Map k v -> Map k v

Inserts a key only if it is not already present — existing values are kept.

insertWith (a => b => a === b) ('a') (9) ([['a', 1]]) // => [['a', 1]]
insertWith (a => b => a === b) ('b') (2) ([['a', 1]]) // => [['a', 1], ['b', 2]]

insertWithCombine :: (k -> k -> Boolean) -> (v -> v -> v) -> k -> v -> Map k v -> Map k v

Inserts, or combines the new value with the existing one using f(newVal)(oldVal).

insertWithCombine (a => b => a === b) (a => b => a + b) ('a') (10) ([['a', 1]])
// => [['a', 11]]

remove :: (k -> k -> Boolean) -> k -> Map k v -> Map k v

Removes the entry with the given key. No-op if absent.

remove (a => b => a === b) ('a') ([['a', 1], ['b', 2]]) // => [['b', 2]]
remove (a => b => a === b) ('z') ([['a', 1]])            // => [['a', 1]]

update :: (k -> k -> Boolean) -> (v -> Maybe v) -> k -> Map k v -> Map k v

Applies f to the value at key if present; removes the entry if f returns Nothing.

update (a => b => a === b) (v => v > 1 ? just(v - 1) : nothing()) ('a') ([['a', 2]])
// => [['a', 1]]
update (a => b => a === b) (v => v > 1 ? just(v - 1) : nothing()) ('a') ([['a', 1]])
// => []

upsert :: (k -> k -> Boolean) -> (v -> v) -> k -> v -> Map k v -> Map k v

Applies f to the value at key if present; inserts val if absent.

upsert (a => b => a === b) (v => v + 1) ('a') (0) ([['a', 5]]) // => [['a', 6]]
upsert (a => b => a === b) (v => v + 1) ('b') (0) ([['a', 5]]) // => [['a', 5], ['b', 0]]

map :: (a -> b) -> Map k a -> Map k b

Applies f to every value, preserving keys and order.

map (v => v * 2) ([['a', 1], ['b', 2]]) // => [['a', 2], ['b', 4]]

mapWithKey :: (k -> a -> b) -> Map k a -> Map k b

Applies f to every key-value pair.

mapWithKey (k => v => `${k}:${v}`) ([['a', 1]]) // => [['a', 'a:1']]

filter :: (v -> Boolean) -> Map k v -> Map k v

Keeps entries whose value satisfies pred.

filter (v => v > 1) ([['a', 1], ['b', 2], ['c', 3]]) // => [['b', 2], ['c', 3]]

filterWithKey :: (k -> v -> Boolean) -> Map k v -> Map k v

Keeps entries whose key and value satisfy pred.

filterWithKey (k => v => k !== 'a' && v > 0) ([['a', 1], ['b', 2]]) // => [['b', 2]]

filterMap :: (a -> Maybe b) -> Map k a -> Map k b

Maps f over values, keeping Just results and discarding Nothing.

filterMap (v => v > 1 ? just(v * 10) : nothing()) ([['a', 1], ['b', 2]])
// => [['b', 20]]

reduce :: (b -> v -> b) -> b -> Map k v -> b

Left fold over values in insertion order with a curried function.

reduce (acc => v => acc + v) (0) ([['a', 1], ['b', 2]]) // => 3

foldWithKey :: (b -> k -> v -> b) -> b -> Map k v -> b

Left fold over key-value pairs in insertion order.

foldWithKey (acc => k => v => [...acc, `${k}=${v}`]) ([]) ([['a', 1], ['b', 2]])
// => ['a=1', 'b=2']

union :: (k -> k -> Boolean) -> Map k v -> Map k v -> Map k v

Right-biased union — keys from b overwrite keys from a.

union (a => b => a === b) ([['a', 1]]) ([['b', 2]])       // => [['a', 1], ['b', 2]]
union (a => b => a === b) ([['a', 1]]) ([['a', 9], ['b', 2]]) // => [['a', 9], ['b', 2]]

unionLeft :: (k -> k -> Boolean) -> Map k v -> Map k v -> Map k v

Left-biased union — keys from a take priority over keys from b.

unionLeft (a => b => a === b) ([['a', 1]]) ([['a', 9], ['b', 2]]) // => [['a', 1], ['b', 2]]

unionWith :: (k -> k -> Boolean) -> (v -> v -> v) -> Map k v -> Map k v -> Map k v

Union with a combining function for duplicate keys: f(aVal)(bVal).

unionWith (a => b => a === b) (a => b => a + b) ([['a', 1]]) ([['a', 2], ['b', 3]])
// => [['a', 3], ['b', 3]]

intersection :: (k -> k -> Boolean) -> Map k a -> Map k b -> Map k b

Keeps only entries whose key appears in both maps; values come from b.

intersection (a => b => a === b) ([['a', 1], ['b', 2]]) ([['b', 9], ['c', 3]])
// => [['b', 9]]

intersectionWith :: (k -> k -> Boolean) -> (a -> b -> c) -> Map k a -> Map k b -> Map k c

Intersection keeping values from a, combined with values from b via f(aVal)(bVal).

intersectionWith (a => b => a === b) (a => b => a + b) ([['a', 1], ['b', 2]]) ([['b', 3]])
// => [['b', 5]]

difference :: (k -> k -> Boolean) -> Map k a -> Map k b -> Map k a

Keeps entries from a whose key does NOT appear in b.

difference (a => b => a === b) ([['a', 1], ['b', 2]]) ([['b', 9]])
// => [['a', 1]]

equals :: (k -> k -> Boolean) -> (v -> v -> Boolean) -> Map k v -> Map k v -> Boolean

True when both Maps contain the same key-value pairs (key equality via eqK,
value equality via eqV), regardless of insertion order.

equals (a => b => a === b) (a => b => a === b) ([['a', 1]]) ([['a', 1]]) // => true
equals (a => b => a === b) (a => b => a === b) ([['a', 1]]) ([['a', 2]]) // => false

keys :: Map k v -> Array k

Returns all keys in insertion order.

keys ([['a', 1], ['b', 2]]) // => ['a', 'b']

values :: Map k v -> Array v

Returns all values in insertion order.

values ([['a', 1], ['b', 2]]) // => [1, 2]

all :: (v -> Boolean) -> Map k v -> Boolean

True when every value satisfies pred.

all (v => v > 0) ([['a', 1], ['b', 2]]) // => true
all (v => v > 1) ([['a', 1], ['b', 2]]) // => false

any :: (v -> Boolean) -> Map k v -> Boolean

True when at least one value satisfies pred.

any (v => v > 1) ([['a', 1], ['b', 2]]) // => true
any (v => v > 9) ([['a', 1], ['b', 2]]) // => false

none :: (v -> Boolean) -> Map k v -> Boolean

True when no value satisfies pred.

none (v => v > 9) ([['a', 1], ['b', 2]]) // => true
none (v => v > 1) ([['a', 1], ['b', 2]]) // => false

Set

Functional Set with arbitrary equality. Immutable, equality-function-based.

empty :: () -> Set a

Returns a new empty Set.

empty () // => []

singleton :: a -> Set a

Creates a Set with a single element.

singleton (1) // => [1]

fromArray :: (a -> a -> Boolean) -> Array a -> Set a

Builds a Set from an array, discarding duplicates (later duplicates are
ignored — first occurrence wins).

fromArray (a => b => a === b) ([1, 2, 1, 3, 2]) // => [1, 2, 3]

toArray :: Set a -> Array a

Converts a Set to a plain array (identity — Sets are already arrays).

toArray ([1, 2, 3]) // => [1, 2, 3]

size :: Set a -> Integer

Returns the number of elements in the Set.

size ([1, 2, 3]) // => 3
size ([])        // => 0

isEmpty :: Set a -> Boolean

True when the Set has no elements.

isEmpty ([])    // => true
isEmpty ([1])   // => false

member :: (a -> a -> Boolean) -> a -> Set a -> Boolean

True when the element is a member of the Set.

member (a => b => a === b) (2) ([1, 2, 3]) // => true
member (a => b => a === b) (9) ([1, 2, 3]) // => false

notMember :: (a -> a -> Boolean) -> a -> Set a -> Boolean

True when the element is NOT a member of the Set.

notMember (a => b => a === b) (9) ([1, 2, 3]) // => true
notMember (a => b => a === b) (1) ([1, 2, 3]) // => false

insert :: (a -> a -> Boolean) -> a -> Set a -> Set a

Returns a new Set with the element added.
If the element is already a member the Set is returned unchanged.

insert (a => b => a === b) (4) ([1, 2, 3]) // => [1, 2, 3, 4]
insert (a => b => a === b) (2) ([1, 2, 3]) // => [1, 2, 3]

remove :: (a -> a -> Boolean) -> a -> Set a -> Set a

Returns a new Set with the element removed.
No-op if the element is not a member.

remove (a => b => a === b) (2) ([1, 2, 3]) // => [1, 3]
remove (a => b => a === b) (9) ([1, 2, 3]) // => [1, 2, 3]

union :: (a -> a -> Boolean) -> Set a -> Set a -> Set a

Union — all elements that appear in either Set.
Elements from a come first; elements from b are appended if not already
present (left-biased on duplicates).

union (a => b => a === b) ([1, 2]) ([2, 3]) // => [1, 2, 3]
union (a => b => a === b) ([1])    ([2, 3]) // => [1, 2, 3]

intersection :: (a -> a -> Boolean) -> Set a -> Set a -> Set a

Intersection — elements that appear in both Sets (values from a).

intersection (a => b => a === b) ([1, 2, 3]) ([2, 3, 4]) // => [2, 3]
intersection (a => b => a === b) ([1, 2])    ([3, 4])    // => []

difference :: (a -> a -> Boolean) -> Set a -> Set a -> Set a

Difference — elements in a that are NOT in b.

difference (a => b => a === b) ([1, 2, 3]) ([2, 3]) // => [1]
difference (a => b => a === b) ([1, 2, 3]) ([4, 5]) // => [1, 2, 3]

symmetricDifference :: (a -> a -> Boolean) -> Set a -> Set a -> Set a

Symmetric difference — elements in exactly one of the two Sets.

symmetricDifference (a => b => a === b) ([1, 2, 3]) ([2, 3, 4]) // => [1, 4]

isSubsetOf :: (a -> a -> Boolean) -> Set a -> Set a -> Boolean

True when every element of a is also in b (a ⊆ b).

isSubsetOf (a => b => a === b) ([1, 2]) ([1, 2, 3]) // => true
isSubsetOf (a => b => a === b) ([1, 4]) ([1, 2, 3]) // => false
isSubsetOf (a => b => a === b) ([])     ([1, 2, 3]) // => true

equals :: (a -> a -> Boolean) -> Set a -> Set a -> Boolean

True when a ⊆ b and b ⊆ a — both Sets contain the same elements.

equals (a => b => a === b) ([1, 2, 3]) ([3, 1, 2]) // => true
equals (a => b => a === b) ([1, 2])    ([1, 2, 3]) // => false

map :: (a -> b) -> Set a -> Set b

Applies f to every element.
Note: the result may contain duplicates if f maps distinct elements to equal
values — pass the result through `fromArray` if uniqueness is required.

map (x => x * 2) ([1, 2, 3]) // => [2, 4, 6]

mapUniq :: (b -> b -> Boolean) -> (a -> b) -> Set a -> Set b

Like map, but re-deduplicates the result using the provided equality.

mapUniq (a => b => a === b) (x => x % 2) ([1, 2, 3, 4]) // => [1, 0]

filter :: (a -> Boolean) -> Set a -> Set a

Keeps elements satisfying pred.

filter (x => x > 1) ([1, 2, 3]) // => [2, 3]

reduce :: (b -> a -> b) -> b -> Set a -> b

Left fold over elements in insertion order with a curried function.

reduce (acc => x => acc + x) (0) ([1, 2, 3]) // => 6

all :: (a -> Boolean) -> Set a -> Boolean

True when every element satisfies pred.

all (x => x > 0) ([1, 2, 3]) // => true
all (x => x > 1) ([1, 2, 3]) // => false

any :: (a -> Boolean) -> Set a -> Boolean

True when at least one element satisfies pred.

any (x => x > 2) ([1, 2, 3]) // => true
any (x => x > 9) ([1, 2, 3]) // => false

none :: (a -> Boolean) -> Set a -> Boolean

True when no element satisfies pred.

none (x => x > 9) ([1, 2, 3]) // => true
none (x => x > 2) ([1, 2, 3]) // => false

find :: (a -> Boolean) -> Set a -> Maybe a

Returns Just the first element satisfying pred, or Nothing.

find (x => x > 2) ([1, 2, 3, 4]) // => just(3)
find (x => x > 9) ([1, 2, 3])    // => nothing()

count :: (a -> Boolean) -> Set a -> Integer

Returns the number of elements satisfying pred.

count (x => x > 1) ([1, 2, 3]) // => 2

partition :: (a -> Boolean) -> Set a -> [Set a, Set a]

Splits the Set into [matching, nonMatching] based on pred.

partition (x => x > 1) ([1, 2, 3]) // => [[2, 3], [1]]
