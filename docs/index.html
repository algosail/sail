pipeWith

 fn
pipeWith :: (f (a->b) -> f a -> f b) -> f a -> Array (a -> f b) -> f b

Threads a value through a sequence of steps where each step is applied via
a custom "bind" combinator. The canonical use-case is ADT pipelines:
pass the ADT's `chain` (or `map`) as the binder so each step can
short-circuit or transform the wrapped value without manual unwrapping.

pipeWith (chain) (just (1)) ([x => just (x + 1), x => just (x * 2)])
is equivalent to:
chain (x => just (x * 2)) (chain (x => just (x + 1)) (just (1)))

// pipeWith :: (f (a->b) -> f a -> f b) -> f a -> Array (a -> f b) -> f b

// Maybe pipeline — short-circuits on Nothing
import * as M from './maybe.js'
pipeWith (M.chain) (M.just (4)) ([
  x => x > 0  ? M.just (x)      : M.nothing (),
  x => x < 10 ? M.just (x * 2)  : M.nothing (),
])
// => just(8)

// map-only pipeline (no short-circuiting)
pipeWith (M.map) (M.just (3)) ([x => x + 1, x => x * 2])
// => just(8)

Like `cond`, but returns Right(fn(a)) on match and Left(a) when no branch

condDefault

 logic
condDefault :: (a -> b) -> Array [(a -> Boolean), (a -> b)] -> a -> b

Like `cond`, but takes an explicit default function instead of returning
Maybe — eliminates the need to unwrap Just/Nothing at every call site.

// condDefault :: (a -> b) -> Array [(a -> Boolean), (a -> b)] -> a -> b
condDefault (x => `unhandled: ${x}`) ([
  [x => x < 0, () => 'negative'],
  [x => x > 0, () => 'positive'],
]) (0)  // => 'unhandled: 0'
condDefault (x => `unhandled: ${x}`) ([
  [x => x < 0, () => 'negative'],
  [x => x > 0, () => 'positive'],
]) (5)  // => 'positive'

match

 logic
match :: Array [(a -> Maybe b), (b -> c)] -> a -> Maybe c

Pattern-matching variant of `cond` — each "predicate" is a function
`a -> Maybe b` that simultaneously tests and extracts data. The first
branch returning Just wins; its inner value is passed to the handler.
This is strictly more powerful than `cond`: use it when the predicate
needs to capture a sub-value (like a regex match or a type-narrowing
accessor) that the handler then consumes.

// match :: Array [(a -> Maybe b), (b -> c)] -> a -> Maybe c
match ([
  [x => x > 0  ? M.just (x)  : M.nothing (), x => `pos: ${x}`],
  [x => x < 0  ? M.just (-x) : M.nothing (), x => `neg abs: ${x}`],
]) (-5)  // => just('neg abs: 5')
match ([
  [x => x > 0  ? M.just (x)  : M.nothing (), x => `pos: ${x}`],
]) (0)   // => nothing()

condAll

 logic
condAll :: Array [(a -> Boolean), (a -> b)] -> a -> Array b

Runs every branch whose predicate matches and collects all results —
unlike `cond`, which stops at the first match. Returns an empty array
when no predicate holds.

// condAll :: Array [(a -> Boolean), (a -> b)] -> a -> Array b
condAll ([
  [x => x > 0,      () => 'positive'],
  [x => x % 2 === 0, () => 'even'],
  [x => x > 100,    () => 'large'],
]) (4)   // => ['positive', 'even']
condAll ([
  [x => x > 0,      () => 'positive'],
]) (-1)  // => []

guardWith

 logic
guardWith :: Array (a -> Boolean | a -> Maybe b) -> a -> Maybe a

Threads a value through a sequence of guard steps, short-circuiting on the
first Nothing. Each step is either:
- a predicate `a -> Boolean` — Nothing when it returns false, Just(a) otherwise
- a transformer `a -> Maybe b` — its result is passed straight through

The two forms are distinguished automatically: if the step returns a boolean
it is treated as a predicate; if it returns a Maybe (has a `.tag`) it is
treated as a transformer.

// guardWith :: Array (a -> Boolean | a -> Maybe b) -> a -> Maybe a
guardWith ([
  x => x > 0,              // predicate: filter non-positives
  x => x < 100,            // predicate: filter >= 100
]) (42)   // => just(42)

guardWith ([
  x => x > 0,
  x => x < 100,
]) (-1)   // => nothing()

guardWith ([
  x => x > 0,
  x => M.just (x * 2),     // transformer: doubles the value
  x => x < 100,
]) (42)   // => just(84)

guardWith ([
  x => x > 0,
  x => M.just (x * 2),
  x => x < 100,
]) (60)   // => nothing()  (60 * 2 = 120, fails x < 100)

concat

concat :: (a -> a -> a) -> (b -> b -> b) -> These a b -> These a b -> These a b

Combines two These values, accumulating both sides with their respective
Semigroup concats. The table (using Haskell notation):

This a <> This b = This (a <> b)
This a <> That y = Both a y
This a <> Both b y = Both (a <> b) y
That x <> This b = Both b x
That x <> That y = That (x <> y)
That x <> Both b y = Both b (x <> y)
Both a x <> This b = Both (a <> b) x
Both a x <> That y = Both a (x <> y)
Both a x <> Both b y = Both (a <> b) (x <> y)

// concat :: (a -> a -> a) -> (b -> b -> b) -> These a b -> These a b -> These a b
const c = concat (a => b => a + b) (a => b => a + b)
c (this_ ('a')) (this_ ('b'))     // => this_('ab')
c (that (1))    (that (2))        // => that(3)
c (this_ ('a')) (that (1))        // => both('a')(1)
c (both ('a') (1)) (both ('b') (2)) // => both('ab')(3)

map

 these
map :: (b -> c) -> These a b -> These a c

Applies f to the right (b) value; passes This through unchanged.

// map :: (b -> c) -> These a b -> These a c
map (x => x + 1) (that (1))        // => that(2)
map (x => x + 1) (both ('w') (1))  // => both('w')(2)
map (x => x + 1) (this_ ('w'))     // => this_('w')

bimap (a => a + '!') (b => b * 2) (this_ ('w'))    // => this_('w!')

 these

ap

 these
ap :: (a -> a -> a) -> These a (b -> c) -> These a b -> These a c

Apply — combines These values, accumulating the left side with concatA.
This is analogous to Validation's ap but operates on These.

// ap :: (a -> a -> a) -> These a (b -> c) -> These a b -> These a c

apStr (that (x => x + 1))        (that (2))         // => that(3)
apStr (this_ ('e1'))              (that (2))         // => this_('e1')
apStr (that (x => x + 1))        (this_ ('e2'))      // => this_('e2')
apStr (this_ ('e1'))              (this_ ('e2'))      // => this_('e1e2')
apStr (both ('w') (x => x + 1))  (that (2))         // => both('w')(3)
apStr (both ('w1') (x => x + 1)) (both ('w2') (2)) // => both('w1w2')(3)

chain

 these
chain :: (a -> a -> a) -> (b -> These a c) -> These a b -> These a c

Monadic bind — threads the left side through using concatA.

const chainStr = chain (a => b => a + b)
chainStr (x => that (x + 1))       (that (2))         // => that(3)
chainStr (x => both ('w2') (x + 1)) (both ('w1') (2)) // => both('w1w2')(3)
chainStr (x => that (x + 1))        (this_ ('e'))      // => this_('e')

fold

 these

Reduces the right (b) value with f and init; returns init for This.

Converts to Maybe — Just for That/Both (the right value), Nothing for This.

Sequential application — runs the function action, then the value action,

chain

 state
chain :: (a -> State s b) -> State s a -> State s b

Sequences two State actions — the second action can depend on the value
produced by the first. The state threads through both.

// chain :: (a -> State s b) -> State s a -> State s b

run (action) (0) // => [null, 1]

Sequences two State actions, discarding the value of the first.
Haskell: (>>)

chainRec

 state
chainRec :: ((a -> Step, b -> Step, a) -> State s Step) -> a -> State s b

Stack-safe tail-recursive State bind (Fantasy Land ChainRec).
f receives (next, done, a) and must return a State action that yields a Step.

// chainRec :: ((a -> Step, b -> Step, a) -> State s Step) -> a -> State s b
// Count down from n, accumulating increments in state:
const action = chainRec ((next, done, n) =>
  n <= 0 ? of (done (0)) : chain ((_) => of (next (n - 1))) (modify (s => s + 1))
) (5)
run (action) (0) // => [0, 5]

sequence

 state
sequence :: Array (State s a) -> State s (Array a)

Runs a list of State actions in sequence, collecting all values.
The state threads through each action left to right.

// sequence :: Array (State s a) -> State s (Array a)
run (sequence ([of (1), of (2), of (3)])) (0) // => [[1, 2, 3], 0]

traverse

 state
traverse :: (a -> State s b) -> Array a -> State s (Array b)

Maps f over an array, running each resulting State action in sequence and

// traverse :: (a -> State s b) -> Array a -> State s (Array b)

lift2

Runs two State actions and combines their values with a curried binary

Functional lenses for immutable data access and update.

 lens

Constructs a Lens from a getter and a curried setter.

Composes two lenses left-to-right: the first lens focuses on a part of `s`,

traverse :: (b -> f b) -> ((a -> b) -> f a -> f b) -> (a -> f b) -> [a, c] -> f [b, c]

Traverses the first element through an applicative functor.
Accepts explicit apOf / apMap to remain functor-agnostic.

apOf :: b -> f b (pure / of)
apMap :: (a -> b) -> f a -> f b (map, curried)

// traverse :: (b -> f b) -> ((a -> b) -> f a -> f b) -> (a -> f b) -> [a, c] -> f [b, c]
const apOf  = Array.of
const apMap = f => xs => xs.map (f)
traverse (apOf) (apMap) (x => [x, -x]) ([1, 2]) // => [[1, 2], [-1, 2]]

True when at least one element satisfies pred.

// any :: (a -> Boolean) -> Set a -> Boolean
any (x => x > 2) ([1, 2, 3]) // => true
any (x => x > 9) ([1, 2, 3]) // => false

find :: (a -> Boolean) -> Set a -> Maybe a

Returns Just the first element satisfying pred, or Nothing.

// find :: (a -> Boolean) -> Set a -> Maybe a
find (x => x > 2) ([1, 2, 3, 4]) // => just(3)
find (x => x > 9) ([1, 2, 3])    // => nothing()

// count :: (a -> Boolean) -> Set a -> Integer

partition :: (a -> Boolean) -> Set a -> [Set a, Set a]

Splits the Set into [matching, nonMatching] based on pred.

// partition :: (a -> Boolean) -> Set a -> [Set a, Set a]
partition (x => x > 1) ([1, 2, 3]) // => [[2, 3], [1]]
